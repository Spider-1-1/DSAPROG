    int linear(int arr[],int size,int key)
    {
    int i,f=0;
    for(i=0;i<size;i++)
    {
    if(arr[i]==key)
    {
    f=1;
    break;
    }
    }
    if(f==1)
    {
    return i;
    }
    else
    {
    return -1;
    }
    }
